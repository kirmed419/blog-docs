Getting Started with Matplotlib

Learn about Matplotlib.
Kirouane Mohammed

Introduction to Matplotlib#

Matplotlib is a powerful and versatile Python library used for creating static, animated, and interactive visualizations. It is the foundation for many other plotting libraries, such as Seaborn and Pandas’ plotting capabilities. Matplotlib’s flexibility allows you to create a wide variety of plots, ranging from simple line graphs to complex multi-plot figures.

Matplotlib primarily operates through two key interfaces: pyplot, which is a state-based interface similar to MATLAB, and an object-oriented interface that gives you more control and customization over your plots.

Whether you’re visualizing simple data or creating intricate visual representations, Matplotlib provides the tools needed to effectively communicate your data’s story.

Matplotlib Guide#

Importing and Setup#

import matplotlib.pyplot as plt
import numpy as np

Basic Plotting#

# Simple line plot
x = np.linspace(0, 10, 100)
y = np.sin(x)

plt.plot(x, y)
plt.show()

Figure and Axes#

# Create a figure and a set of subplots (axes)
fig, ax = plt.subplots()

# Multiple subplots
fig, axs = plt.subplots(2, 2)  # 2x2 grid of subplots

Customizing Plots#

# Title and labels
plt.plot(x, y)
plt.title('Sine Wave')
plt.xlabel('X axis')
plt.ylabel('Y axis')

# Grid
plt.grid(True)

# Limits
plt.xlim(0, 10)
plt.ylim(-1, 1)

# Legends
plt.plot(x, y, label='Sine')
plt.plot(x, np.cos(x), label='Cosine')
plt.legend()

# Saving a plot
plt.savefig('plot.png')

Plot Types#

Line Plot#

plt.plot(x, y)

Scatter Plot#

x = np.random.rand(50)
y = np.random.rand(50)

plt.scatter(x, y)

Bar Plot#

categories = ['A', 'B', 'C']
values = [1, 4, 2]

plt.bar(categories, values)

Histogram#

data = np.random.randn(1000)

plt.hist(data, bins=30)

Box Plot#

data = [np.random.randn(100) for _ in range(4)]

plt.boxplot(data)

Pie Chart#

labels = ['A', 'B', 'C']
sizes = [50, 25, 25]

plt.pie(sizes, labels=labels, autopct='%1.1f%%')

Heatmap#

data = np.random.rand(10, 10)

plt.imshow(data, cmap='hot', interpolation='nearest')
plt.colorbar()

Customizing Appearance#

Line Styles and Colors#

plt.plot(x, y, color='red', linestyle='--', linewidth=2)

Markers#

plt.plot(x, y, marker='o', markersize=5, markerfacecolor='blue')

Customizing Ticks#

plt.plot(x, y)
plt.xticks(np.arange(0, 11, step=1))
plt.yticks(np.arange(-1, 1.5, step=0.5))

Subplot Adjustments#

fig, axs = plt.subplots(2, 1)
plt.subplots_adjust(hspace=0.5, wspace=0.5)

Adding Text#

plt.plot(x, y)
plt.text(5, 0, 'Center', fontsize=12, ha='center')

Working with Dates#

import matplotlib.dates as mdates
import datetime

dates = [datetime.datetime(2023, 1, i+1) for i in range(10)]
values = np.random.randn(10)

plt.plot(dates, values)
plt.gcf().autofmt_xdate()  # Rotate dates
plt.gca().xaxis.set_major_formatter(mdates.DateFormatter('%Y-%m-%d'))

Logarithmic and Polar Plots#

# Logarithmic scale
plt.plot(x, y)
plt.yscale('log')

# Polar plot
theta = np.linspace(0, 2 * np.pi, 100)
r = np.abs(np.sin(theta))

plt.polar(theta, r)

Advanced Features#

Twin Axes#

fig, ax1 = plt.subplots()

ax2 = ax1.twinx()
ax1.plot(x, y, 'g-')
ax2.plot(x, np.cos(x), 'b-')

ax1.set_xlabel('X data')
ax1.set_ylabel('Sine', color='g')
ax2.set_ylabel('Cosine', color='b')

Inset Plots#

fig, ax = plt.subplots()
ax.plot(x, y)

# Inset plot
ax_inset = fig.add_axes([0.5, 0.5, 0.4, 0.4])
ax_inset.plot(x, np.cos(x))

3D Plotting#

from mpl_toolkits.mplot3d import Axes3D

fig = plt.figure()
ax = fig.add_subplot(111, projection='3d')

x = np.linspace(-5, 5, 100)
y = np.linspace(-5, 5, 100)
X, Y = np.meshgrid(x, y)
Z = np.sin(np.sqrt(X**2 + Y**2))

ax.plot_surface(X, Y, Z, cmap='viridis')

Styles and Themes#

# Use a built-in style
plt.style.use('ggplot')

# List available styles
plt.style.available

Interactive Plotting with plt.show()#

# Interactive mode
plt.ion()

# Non-blocking show
plt.show(block=False)

# Turn off interactive mode
plt.ioff()

Annotating Plots#

plt.plot(x, y)
plt.annotate('Max Value', xy=(np.pi/2, 1), xytext=(np.pi, 0.5),
             arrowprops=dict(facecolor='black', shrink=0.05))

Configuring Matplotlib#

# Set default figure size
plt.rcParams['figure.figsize'] = [10, 5]

# Set default DPI
plt.rcParams['figure.dpi'] = 100

# Change font size
plt.rcParams['font.size'] = 12

Saving Plots#

# Save figure
plt.savefig('plot.png')

# Save with different DPI
plt.savefig('plot.png', dpi=300)

# Save in different formats (e.g., PDF)
plt.savefig('plot.pdf')
