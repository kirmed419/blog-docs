Introduction to Pandas: The Python Data Analysis Powerhouse#

Pandas is a powerful open-source data analysis and manipulation library built on top of Python. If you’re familiar with Excel and looking to dive into more robust and scalable data analysis, pandas is a natural next step. While Excel is excellent for small datasets and manual data manipulation, pandas shines when it comes to handling large datasets, automating repetitive tasks, and performing complex data manipulations with just a few lines of code.

Why Use Pandas Instead of Excel?#

  1. Scalability: Pandas can handle much larger datasets than Excel, which tends to slow down or crash when dealing with massive amounts of data.
  2. Automation: Repetitive tasks that require manual effort in Excel can be automated in pandas using scripts, saving time and reducing the risk of human error.
  3. Flexibility: Pandas offers more flexibility with data manipulation, allowing for complex operations such as merging, reshaping, and filtering data that would be cumbersome in Excel.
  4. Integration: Pandas integrates seamlessly with other Python libraries such as NumPy, Matplotlib, and SciPy, enabling advanced data analysis and visualization that Excel cannot match.
  5. Reproducibility: Pandas scripts can be saved, shared, and rerun on different datasets, ensuring that your analysis is reproducible.

Pandas User Guide#

Importing and Setup#

import pandas as pd
import numpy as np

Creating DataFrames#

# From a dictionary
data = {'col1': [1, 2], 'col2': [3, 4]}
df = pd.DataFrame(data)

# From a list of dictionaries
data = [{'a': 1, 'b': 2}, {'a': 3, 'b': 4}]
df = pd.DataFrame(data)

# From a NumPy array
data = np.array([[1, 2], [3, 4]])
df = pd.DataFrame(data, columns=['col1', 'col2'])

Reading/Writing Data#

# Reading CSV
df = pd.read_csv('file.csv')

# Writing CSV
df.to_csv('file.csv', index=False)

# Reading Excel
df = pd.read_excel('file.xlsx')

# Writing Excel
df.to_excel('file.xlsx', index=False)

# Reading JSON
df = pd.read_json('file.json')

# Writing JSON
df.to_json('file.json')

Data Inspection#

# Show top/bottom rows
df.head()    # First 5 rows
df.tail()    # Last 5 rows

# Data info
df.info()    # Summary of the DataFrame
df.describe() # Descriptive statistics

# Data types
df.dtypes

# Shape of DataFrame
df.shape     # (rows, columns)

Selecting Data#

# Selecting columns
df['col1']           # Single column (returns Series)
df[['col1', 'col2']] # Multiple columns (returns DataFrame)

# Selecting rows by index
df.iloc[0]           # First row
df.iloc[-1]          # Last row

# Selecting rows by label
df.loc[0]            # First row (if labeled)
df.loc[df['col1'] > 1] # Rows where condition is met

# Selecting specific value
df.at[0, 'col1']      # Value at row 0, column 'col1'
df.iat[0, 0]          # Value at first row and first column

Filtering Data#

# Conditional filtering
df[df['col1'] > 1]

# Filtering with multiple conditions
df[(df['col1'] > 1) & (df['col2'] < 4)]

# Filtering using `query`
df.query('col1 > 1 and col2 < 4')

Modifying DataFrames#

# Adding a new column
df['col3'] = df['col1'] + df['col2']

# Renaming columns
df.rename(columns={'col1': 'new_col1'}, inplace=True)

# Dropping columns
df.drop(columns=['col1'], inplace=True)

# Dropping rows
df.drop([0, 1], inplace=True)  # Drop rows by index

# Replacing values
df['col1'].replace(1, 100, inplace=True)

# Applying functions
df['col1'] = df['col1'].apply(lambda x: x * 2)

# Changing the order of columns
df = df[['col2', 'col3', 'col1']]

Handling Missing Data#

# Check for missing data
df.isnull()       # Returns DataFrame of booleans
df.isnull().sum() # Number of missing values per column

# Drop missing data
df.dropna()       # Drop rows with any missing data
df.dropna(axis=1) # Drop columns with any missing data

# Fill missing data
df.fillna(0)      # Fill missing values with 0
df['col1'].fillna(df['col1'].mean(), inplace=True) # Fill with mean

Grouping and Aggregation#

# Group by a column and aggregate
df.groupby('col1').sum()     # Sum all columns grouped by 'col1'
df.groupby('col1')['col2'].mean() # Mean of 'col2' grouped by 'col1'

# Multi-level grouping
df.groupby(['col1', 'col2']).size()

# Aggregate using different functions
df.groupby('col1').agg({'col2': 'mean', 'col3': 'sum'})

Merging and Joining#

# Merging DataFrames
df1 = pd.DataFrame({'key': ['A', 'B', 'C'], 'col1': [1, 2, 3]})
df2 = pd.DataFrame({'key': ['B', 'C', 'D'], 'col2': [4, 5, 6]})

# Inner join (only keys present in both)
merged = pd.merge(df1, df2, on='key', how='inner')

# Outer join (all keys, fill missing with NaN)
merged = pd.merge(df1, df2, on='key', how='outer')

# Left join (all keys from df1)
merged = pd.merge(df1, df2, on='key', how='left')

# Right join (all keys from df2)
merged = pd.merge(df1, df2, on='key', how='right')

# Concatenating DataFrames
df = pd.concat([df1, df2], axis=0)  # Stack rows
df = pd.concat([df1, df2], axis=1)  # Stack columns

Sorting and Ranking#

# Sorting by column
df.sort_values('col1', ascending=False)

# Sorting by multiple columns
df.sort_values(['col1', 'col2'], ascending=[True, False])

# Ranking
df['rank'] = df['col1'].rank()

Pivoting and Reshaping#

# Pivot table
df.pivot_table(values='col1', index='col2', columns='col3', aggfunc=np.mean)

# Melting (unpivot)
df_melted = pd.melt(df, id_vars=['col1'], value_vars=['col2', 'col3'])

# Reshape with stack/unstack
stacked = df.stack()    # Convert columns to rows
unstacked = stacked.unstack()  # Convert rows back to columns

Time Series#

# Convert column to datetime
df['date'] = pd.to_datetime(df['date'])

# Setting a datetime index
df.set_index('date', inplace=True)

# Resampling time series data
df.resample('M').mean()  # Resample by month

Input/Output Performance#

# Reading large CSV with specific types
df = pd.read_csv('file.csv', dtype={'col1': 'int32', 'col2': 'float32'})

# Using chunksize to process large files
for chunk in pd.read_csv('file.csv', chunksize=10000):
    process(chunk)  # process() is a placeholder function

DataFrame Operations#

# Transpose
df.T

# Vectorized operations
df['col1'] * 2   # Multiply entire column by 2
df['col1'] + df['col2']  # Add two columns element-wise

# Apply function across DataFrame
df.apply(np.sqrt)  # Apply function element-wise
df.applymap(lambda x: x**2)  # Apply function to each element

Miscellaneous#

# Duplicates
df.duplicated()          # Find duplicates
df.drop_duplicates()     # Drop duplicate rows

# Memory usage
df.memory_usage()

# Random sampling
df.sample(5)             # Randomly sample 5 rows

# Convert to other formats
df.to_dict()             # Convert to dictionary
df.to_numpy()            # Convert to NumPy array

# Resetting index
df.reset_index(drop=True, inplace=True)
